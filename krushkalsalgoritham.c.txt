#include <stdio.h>
#include <stdlib.h>

#define MAX 10
#define INF 999

int parent[MAX];

// Initialize all parent values to 0
void initialize(int n) {
    for (int i = 1; i <= n; i++)
        parent[i] = 0;
}

// Find function with path compression
int find(int i) {
    while (parent[i] != 0)
        i = parent[i];
    return i;
}

// Union function
int uni(int i, int j) {
    if (i != j) {
        parent[j] = i;
        return 1;
    }
    return 0;
}

int main() {
    int vertex_count;
    int cost_matrix[MAX][MAX];
    int edge_count, count = 1;
    int sum_cost = 0, min_cost;
    int row_no, column_no, edge1 = 0, edge2 = 0;

    printf("Implementation of Kruskal's Algorithm\n");
    printf("Enter total number of vertices: ");
    scanf("%d", &vertex_count);

    printf("Enter the cost adjacency matrix (0 for no edge):\n");
    for (int row = 1; row <= vertex_count; row++) {
        for (int column = 1; column <= vertex_count; column++) {
            scanf("%d", &cost_matrix[row][column]);
            if (cost_matrix[row][column] == 0)
                cost_matrix[row][column] = INF; // Replace 0 with INF
        }
    }

    initialize(vertex_count);
    edge_count = vertex_count - 1;

    while (count <= edge_count) {
        min_cost = INF;
        for (int row = 1; row <= vertex_count; row++) {
            for (int column = 1; column <= vertex_count; column++) {
                if (cost_matrix[row][column] < min_cost) {
                    min_cost = cost_matrix[row][column];
                    edge1 = row_no = row;
                    edge2 = column_no = column;
                }
            }
        }

        row_no = find(row_no);
        column_no = find(column_no);

        if (uni(row_no, column_no)) {
            printf("Edge %d: (%d â†’ %d) cost = %d\n", count++, edge1, edge2, min_cost);
            sum_cost += min_cost;
        }

        cost_matrix[edge1][edge2] = cost_matrix[edge2][edge1] = INF;
    }

    printf("\nMinimum cost of spanning tree = %d\n", sum_cost);
    return 0;
}
