#include <stdio.h>
#include <stdlib.h>

#define MAX 50

int parent[MAX];
int rankArr[MAX];

void makeSet(int x) {
    if (parent[x] != -1) {
        printf("\nElement %d is already in a set\n", x);
        return;
    }
    parent[x] = x;
    rankArr[x] = 0;
    printf("Set created for element %d\n", x);
}

int findSet(int x) {
    if (parent[x] == -1) {
        printf("Element %d is not in any set\n", x);
        return -1;
    }
    if (parent[x] != x) {
        parent[x] = findSet(parent[x]); // Path Compression
    }
    return parent[x];
}

void unionSets(int x, int y) {
    int repX = findSet(x);
    int repY = findSet(y);

    if (repX == -1 || repY == -1) return; // Fix: use logical OR (||)

    if (repX != repY) {
        if (rankArr[repX] > rankArr[repY]) {
            parent[repY] = repX;
        } else if (rankArr[repX] < rankArr[repY]) {
            parent[repX] = repY;
        } else {
            parent[repY] = repX;
            rankArr[repX]++;
        }
        printf("Sets containing %d and %d have been merged\n", x, y);
    } else {
        printf("Elements %d and %d are already in the same set\n", x, y);
    }
}

void displaySets(int n) {
    printf("\nElement : Representative\n");
    for (int i = 0; i < n; i++) {
        if (parent[i] != -1) {
            printf("%d : %d\n", i, findSet(i));
        }
    }
}

int main() {
    int n = MAX;
    for (int i = 0; i < n; i++) {
        parent[i] = -1;
    }

    int choice, x, y;
    do {
        printf("\n----- Disjoint Set Operations -----\n");
        printf("1. Make Set\n");
        printf("2. Find Set\n");
        printf("3. Union Sets\n");
        printf("4. Display All Sets\n");
        printf("5. Exit\n");
        printf("Enter your choice: ");
        scanf("%d", &choice);

        switch (choice) {
            case 1:
                printf("Enter the element to create a set: ");
                scanf("%d", &x);
                makeSet(x);
                break;

            case 2:
                printf("Enter the element to find its representative: ");
                scanf("%d", &x);
                y = findSet(x);
                if (y != -1)
                    printf("The representative of %d is %d\n", x, y);
                break;

            case 3:
                printf("Enter the first element: ");
                scanf("%d", &x);
                printf("Enter the second element: ");
                scanf("%d", &y);
                unionSets(x, y);
                break;

            case 4:
                displaySets(n);
                break;

            case 5:
                exit(0);

            default:
                printf("Invalid choice! Please try again.\n");
                break;
        }
    } while (1);

    return 0;
}
